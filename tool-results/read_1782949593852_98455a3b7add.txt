   122→function downloadFile(filename: string, content: string, mime: string) {
   123→  const blob = new Blob([content], { type: mime });
   124→  const url = URL.createObjectURL(blob);
   125→  const a = document.createElement("a");
   126→  a.href = url;
   127→  a.download = filename;
   128→  document.body.appendChild(a);
   129→  a.click();
   130→  document.body.removeChild(a);
   131→  URL.revokeObjectURL(url);
   132→}

   134→async function downloadImage(url: string, filename: string) {
   135→  try {
   136→    const res = await fetch(url);
   137→    const blob = await res.blob();
   138→    const objUrl = URL.createObjectURL(blob);
   139→    const a = document.createElement("a");

   141→    a.download = filename;
   142→    document.body.appendChild(a);
   143→    a.click();

   145→    URL.revokeObjectURL(objUrl);
   146→  } catch {
   147→    toast.error("Could not download image");
   148→  }
