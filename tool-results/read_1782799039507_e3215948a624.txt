    73→    73→function downloadFile(filename: string, content: string, mime: string) {
    74→    74→  const blob = new Blob([content], { type: mime });
    75→    75→  const url = URL.createObjectURL(blob);
    76→    76→  const a = document.createElement("a");
    77→    77→  a.href = url;
    78→    78→  a.download = filename;
    79→    79→  document.body.appendChild(a);
    80→    80→  a.click();
    81→    81→  document.body.removeChild(a);
    82→    82→  URL.revokeObjectURL(url);
    83→    83→}
