    42→function sizeForAspect(aspect: string): string {
    43→  const found = ASPECTS.find((a) => a.id === aspect);
    44→  if (found && isSupportedSize(found.size)) return found.size;
    45→  return "1024x1024";
    46→}

    48→/** Parse "WxH" → { width, height }. Falls back to 1024x1024. */
    49→function parseSize(size: string): { width: number; height: number } {
    50→  const m = /^(\d+)x(\d+)$/.exec(size);
    51→  if (m) return { width: Number(m[1]), height: Number(m[2]) };
    52→  return { width: 1024, height: 1024 };
    53→}

    63→// Robustly extract a JSON object from an LLM response that may contain
    64→// prose / markdown fences around the JSON.
    65→function extractJson<T = unknown>(raw: string): T | null {
    66→  if (!raw) return null;
    67→  // strip code fences
    68→  let s = raw.trim();
    69→  const fence = s.match(/```(?:json)?\s*([\s\S]*?)```/i);
    70→  if (fence) s = fence[1].trim();
    71→  // find first { ... last }
    72→  const first = s.indexOf("{");
    73→  const last = s.lastIndexOf("}");
    74→  if (first !== -1 && last !== -1 && last > first) {
    75→    s = s.slice(first, last + 1);
    76→  }
    77→  // also try first [ ... last ]
    78→  if (first === -1) {
    79→    const a = s.indexOf("[");
    80→    const b = s.lastIndexOf("]");
    81→    if (a !== -1 && b !== -1 && b > a) s = s.slice(a, b + 1);
    82→  }
    83→  try {
    84→    return JSON.parse(s) as T;
    85→  } catch {
    86→    return null;
    87→  }
    88→}

    90→function clamp(n: unknown, lo = 0, hi = 100): number {
    91→  const v = typeof n === "number" ? n : typeof n === "string" ? parseFloat(n) : NaN;
    92→  if (Number.isNaN(v)) return lo;
    93→  return Math.max(lo, Math.min(hi, v));
    94→}
    95→
    96→function strArr(v: unknown): string[] {
    97→  if (Array.isArray(v)) return v.map((x) => String(x)).filter(Boolean);
    98→  if (typeof v === "string" && v.trim()) return [v.trim()];
    99→  return [];
   100→}

   102→// ----------------------------------------------------------------------------
   103→// Stage 1 — FLUX image generation
   104→// ----------------------------------------------------------------------------
   105→// Two backends, selected by MODAL_USE env var:
   106→//   • Modal (default when MODAL_USE=true): calls the user's deployed
   107→//     `nexus-visual` Modal app → real FLUX.1-schnell on H100 GPU.
   108→//     Warm calls ~1.5–2s; cold start up to ~7min.
   109→//   • z-ai-web-dev-sdk (fallback): calls zai.images.generations.create.
   110→//     No GPU, hosted inference, ~20–30s per call.
   111→// ----------------------------------------------------------------------------
   112→export async function stageFlux(
   113→  prompt: string,
   114→  style: string,
   115→  aspect: string,
   116→  wardrobe: string | null,
   117→  generationId: string,
   118→  calibration: ResolvedCalibration,
   119→  loraIds: string[],
   120→  loraWeights: Record<string, number>,
   121→  engineId?: string
   122→): Promise<{ imagePath: string; imageBase64: string; size: string; ms: number; backend: "modal" | "zai"; engineId: string; backendMismatch: boolean }> {
   123→  const start = nowMs();
   124→  const selectedEngine = getEngine(engineId);
   125→
   126→  // Resolve the target size: prefer the calibration resolution when supported,
   127→  // otherwise fall back to the aspect-derived size.
   128→  const aspectSize = sizeForAspect(aspect);
   129→  const size = isSupportedSize(calibration.resolution) ? calibration.resolution : aspectSize;
   130→
   131→  // Build the enriched prompt: base intent + style + wardrobe + LoRA triggers
   132→  // + the preset's quality tokens (these carry the CFG/sampler intent for the
   133→  // Modal backend with real diffusion params).
   134→  const loraTriggers = loraIds
   135→    .map((id) => getLora(id))
   136→    .filter(Boolean)
   137→    .map((l) => l!.name.toLowerCase());
   138→
   139→  const enriched = [
   140→    prompt,
   141→    `style: ${style}`,
   142→    wardrobe ? `wardrobe: ${wardrobe}` : "",
   143→    loraTriggers.length ? `lora triggers: ${loraTriggers.join(", ")}` : "",
   144→    ...calibration.qualityTokens,
   145→  ]
   146→    .filter(Boolean)
   147→    .join(", ");
   148→
   149→  ensureGalleryDir();
   150→  const filename = `${generationId}.png`;
   151→  const filepath = path.join(GALLERY_DIR, filename);
   152→
   153→  let backend: "modal" | "zai" = "modal";
   154→  let base64: string | null = null;
   155→
   156→  // Build the LoRA array for Modal: map our library IDs to HF repo IDs + weights
   157→  const modalLoras = loraIds
   158→    .map((id) => {
   159→      const lora = getLora(id);
   160→      if (!lora) return null;
   161→      // Extract HF repo ID from the URL: https://huggingface.co/NO8D/BodyControl → NO8D/BodyControl
   162→      const repoMatch = lora.url.match(/huggingface\.co\/([^\/]+\/[^\/\?#]+)/);
   163→      if (!repoMatch) return null;
   164→      return {
   165→        repo: repoMatch[1],
   166→        adapter: lora.id, // use our id as the adapter name
   167→        weight: loraWeights[id] ?? lora.recommendedWeight,
   168→      };
   169→    })
   170→    .filter((l): l is { repo: string; adapter: string; weight: number } => l !== null);
   171→
   172→  // Modal is the PRIMARY and ONLY image generation path.
   173→  // The H100 serves FLUX.1-schnell with LoRA support.
   174→  // Cold starts take 10-60s; warm calls return in 2-5s.
   175→  if (isModalEnabled()) {
   176→    try {
   177→      const { width, height } = parseSize(size);
   178→      const result = await generateImageViaModal({
   179→        prompt: enriched,
   180→        width,
   181→        height,
   182→        steps: calibration.steps,
   183→        cfg: calibration.cfg,
   184→        loras: modalLoras,
   185→        isFirstCall: true, // allow long cold-start timeout (300s)
   186→      });
   187→      base64 = result.imageBase64;
   188→      backend = "modal";
   189→      await logEvent(
   190→        "stage_complete",
   191→        `Modal /generate ok — ${result.ms}ms (round-trip ${result.latencyMs}ms) · engine=${selectedEngine.shortName} steps=${calibration.steps} cfg=${calibration.cfg} loras=${modalLoras.length}`,
   192→        "success",
   193→        generationId
   194→      );
   195→    } catch (modalErr) {
   196→      const msg = modalErr instanceof Error ? modalErr.message : String(modalErr);
   197→      await logEvent(
   198→        "error",
   199→        `Modal /generate failed: ${truncate(msg, 300)}`,
   200→        "error",
   201→        generationId
   202→      );
   203→      // Modal is the ONLY image generation path.
   204→      // If Modal fails, throw a clear error so the user knows to retry or warm up.
   205→      throw new Error(`Modal GPU generation failed: ${msg}. The GPU may be cold-starting (wait 30-60s and retry) or the LoRAs may have loading errors. Click "Warm up" first, then try again.`);
   206→    }
   207→  } else {
   208→    throw new Error("MODAL_USE is not true. Modal GPU is the only generation path. Set MODAL_USE=true in .env.");
   209→  }
   210→
   211→  if (!base64) throw new Error("Image generation returned no base64 payload");
   212→
   213→  fs.writeFileSync(filepath, Buffer.from(base64, "base64"));
   214→
   215→  // Detect backend mismatch: the user selected an engine (e.g. Krea 2 Turbo)
   216→  // but the Modal endpoint serves FLUX.1-schnell. This is recorded as
   217→  // provenance so the user knows the actual model used ≠ the selected one.
   218→  // When they deploy the optimized app from Cost Lab, the endpoint will serve
   219→  // the selected engine and this mismatch disappears.
   220→  const backendMismatch = backend === "modal" && selectedEngine.id !== "flux1-schnell" && selectedEngine.family !== "FLUX.1";
   221→  if (backendMismatch) {
   222→    await logEvent(
   223→      "warn",
   224→      `Backend mismatch: selected engine "${selectedEngine.name}" but Modal endpoint serves FLUX.1-schnell. The generated image uses FLUX.1-schnell weights. Deploy the optimized app from Cost Lab to serve ${selectedEngine.name}.`,
   225→      "warn",
   226→      generationId
   227→    );
   228→  }
   229→
   230→  return {
   231→    imagePath: `/gallery/${filename}`,
   232→    imageBase64: base64, // store in DB so images persist across dev server restarts
   233→    size,
   234→    ms: elapsed(start),
   235→    backend,
   236→    engineId: selectedEngine.id,
   237→    backendMismatch,
   238→  };
   239→}

   244→export async function stageSt3gg(
   245→  prompt: string,
   246→  style: string,
   247→  wardrobe: string | null,
   248→  brainId?: string
   249→): Promise<SafetyResult> {
   250→  const start = nowMs();
   251→  const brain = getBrain(brainId);
   252→  const sys = BRAIN_ROLE_PROMPTS.safety + " (Running on " + brain.shortName + ".)";
   253→  const user = `Evaluate this image generation request for safety.
   254→Prompt: "${prompt}"
   255→Style: ${style}
   256→Wardrobe notes: ${wardrobe || "none"}
   257→
   258→Consider: violence, explicit sexual content, real-person likeness abuse, self-harm,
   259→hate symbols, CSAM risk, and wardrobe/undress risk. Score 0-100 where 100 is perfectly safe.
   260→
   261→Emit flags from this catalogue when relevant (use these exact slugs):
   262→- HARD BLOCK (always refused): csam, minors, underage, nonconsensual, deepfake-real-person, real-person-likeness, extreme-violence, gore, hate-symbol, self-harm, bestiality, terrorism
   263→- POLICY CATEGORIES: explicit-nudity, suggestual, wardrobe-risk, violence-mild, violence-graphic, weapons, substance, political, religious
   264→
   265→Respond as JSON exactly in this shape:
   266→{
   267→  "passed": boolean,            // true if risk is low/medium or below and no critical flags
   268→  "score": number,              // 0-100 safety score
   269→  "riskLevel": "safe"|"low"|"medium"|"high"|"critical",
   270→  "flags": string[],            // flag codes from the catalogue above
   271→  "rationale": string           // one short sentence
   272→}`;
   273→
   274→  const _result = await brainChat([
   275→      
   276→      { role: "assistant", content: sys },
   277→      { role: "user", content: user },
   278→    
   279→    ], { temperature: 0.3 });
   280→  const raw = _result.content;
   281→  const parsed = extractJson<{
   282→    passed?: boolean;
   283→    score?: number;
   284→    riskLevel?: string;
   285→    flags?: unknown;
   286→    rationale?: string;
   287→  }>(raw);
   288→
   289→  const score = clamp(parsed?.score ?? 60);
   290→  const riskLevel = (parsed?.riskLevel ?? "low") as SafetyResult["riskLevel"];
   291→  const passed =
   292→    parsed?.passed ?? (riskLevel === "safe" || riskLevel === "low" || riskLevel === "medium");
   293→
   294→  return {
   295→    passed: !!passed,
   296→    score,
   297→    riskLevel,
   298→    flags: strArr(parsed?.flags),
   299→    rationale: parsed?.rationale ?? "No rationale provided.",
   300→    stageMs: elapsed(start),
   301→  };
   302→}

   304→// ----------------------------------------------------------------------------
   305→// Stage 3 — Visual judge (Gemma 4 12B uncensored, vision on generated image)
   306→// ----------------------------------------------------------------------------
   307→export async function stageJudge(
   308→  imagePath: string,
   309→  prompt: string,
   310→  style: string,
   311→  wardrobe: string | null,
   312→  brainId?: string
   313→): Promise<JudgeResult> {
   314→  const start = nowMs();
   315→  const brain = getBrain(brainId);
   316→
   317→  // Read the generated PNG from disk and base64-encode it
   318→  const abs = path.join(process.cwd(), "public", imagePath);
   319→  const buf = fs.existsSync(abs) ? fs.readFileSync(abs) : null;
   320→  if (!buf) throw new Error(`Judge stage: image not found at ${abs}`);
   321→  const dataUrl = `data:image/png;base64,${buf.toString("base64")}`;
   322→
   323→  const sys = BRAIN_ROLE_PROMPTS.judge + " (Running on " + brain.shortName + ".)";
   324→
   325→  const user = `You are judging a generated image.
   326→Original prompt: "${prompt}"
   327→Intended style: ${style}
   328→Wardrobe notes: ${wardrobe || "none"}
   329→
   330→Score each dimension 0-100:
   331→- promptAdherence: how faithfully the image matches the prompt
   332→- visualQuality: sharpness, artifacts, coherence
   333→- aestheticScore: composition, color, appeal
   334→- safetyScore: presence of unsafe content (100 = clean)
   335→- wardrobeMatch: how well wardrobe notes (if any) are reflected
   336→- overallScore: weighted overall
   337→
   338→Then set verdict: "approved" (overall >= 70 and safety >= 80), "rejected" (overall < 45 or safety < 50), else "needs_review".
   339→Provide 2-4 short observations, strengths and weaknesses arrays.
   340→
   341→Respond as JSON exactly:
   342→{
   343→  "promptAdherence": number,
   344→  "visualQuality": number,
   345→  "aestheticScore": number,
   346→  "safetyScore": number,
   347→  "wardrobeMatch": number,
   348→  "overallScore": number,
   349→  "verdict": "approved"|"rejected"|"needs_review",
   350→  "observations": string[],
   351→  "strengths": string[],
   352→  "weaknesses": string[]
   353→}`;
   354→
   355→  const _vresult = await brainVision([
   356→      { role: "assistant", content: sys },
   357→      { role: "user", content: user, image: dataUrl },
   358→    ], { temperature: 0.3 });
   359→
   360→  const raw = _vresult.content;
   361→  const parsed = extractJson<{
   362→    promptAdherence?: number;
   363→    visualQuality?: number;
   364→    aestheticScore?: number;
   365→    safetyScore?: number;
   366→    wardrobeMatch?: number;
   367→    overallScore?: number;
   368→    verdict?: string;
   369→    observations?: unknown;
   370→    strengths?: unknown;
   371→    weaknesses?: unknown;
   372→  }>(raw);
   373→
   374→  const overall = clamp(parsed?.overallScore ?? 0);
   375→  const safety = clamp(parsed?.safetyScore ?? 0);
   376→  let verdict = parsed?.verdict as JudgeResult["verdict"];
   377→  if (verdict !== "approved" && verdict !== "rejected" && verdict !== "needs_review") {
   378→    verdict = overall >= 70 && safety >= 80 ? "approved" : overall < 45 || safety < 50 ? "rejected" : "needs_review";
   379→  }
   380→
   381→  return {
   382→    promptAdherence: clamp(parsed?.promptAdherence),
   383→    visualQuality: clamp(parsed?.visualQuality),
   384→    aestheticScore: clamp(parsed?.aestheticScore),
   385→    safetyScore: safety,
   386→    wardrobeMatch: clamp(parsed?.wardrobeMatch),
   387→    overallScore: overall,
   388→    verdict,
   389→    observations: strArr(parsed?.observations),
   390→    strengths: strArr(parsed?.strengths),
   391→    weaknesses: strArr(parsed?.weaknesses),
   392→    stageMs: elapsed(start),
   393→  };
   394→}

   529→export async function runPipeline(
   530→  input: PipelineRunInput
   531→): Promise<PipelineRunOutput> {
   532→  const timings: TimingMap = {};
   533→  const promptStart = nowMs();
   534→
   535→  // Resolve calibration up-front (so even a blocked run records what was intended).
   536→  const calibration = resolveCalibration(
   537→    input.calibrationId,
   538→    input.calibrationOverrides as Partial<CalibrationPreset> | undefined
   539→  );
   540→
   541→  // 0. create the Generation row (pending) with calibration + lora provenance
   542→  const gen = await db.generation.create({
   543→    data: {
   544→      prompt: input.prompt,
   545→      style: input.style,
   546→      aspect: input.aspect,
   547→      wardrobe: input.wardrobe,
   548→      size: isSupportedSize(calibration.resolution) ? calibration.resolution : sizeForAspect(input.aspect),
   549→      status: "running",
   550→      calibrationId: input.calibrationId,
   551→      calibration: JSON.stringify(calibration),
   552→      loraIds: input.loraIds.join(","),
   553→    },
   554→  });
   555→  await logEvent("pipeline_start", `Pipeline started for "${truncate(input.prompt, 60)}" · engine=${input.engineId ?? "flux2-klein-9b"} preset=${input.calibrationId} brain=${input.brainId ?? "gemma4"} loras=${input.loraIds.length}${input.videoEnabled ? " +video" : ""}`, "info", gen.id);
   556→
   557→  try {
   558→    // Stage: ST3GG safety scan — RUN FIRST so blocked prompts never reach the GPU.
   559→    const safety = await stageSt3gg(input.prompt, input.style, input.wardrobe, input.brainId);
   560→    timings.st3gg = safety.stageMs;
   561→    await db.safetyScan.create({
   562→      data: {
   563→        generationId: gen.id,
   564→        passed: safety.passed,
   565→        score: safety.score,
   566→        riskLevel: safety.riskLevel,
   567→        flags: JSON.stringify(safety.flags),
   568→        rationale: safety.rationale,
   569→        stageMs: safety.stageMs,
   570→      },
   571→    });
   572→    await logEvent(
   573→      "stage_complete",
   574→      `ST3GG safety scan: ${safety.riskLevel} (score ${safety.score}) flags=[${safety.flags.join(",")}]`,
   575→      safety.passed ? "success" : "warn",
   576→      gen.id
   577→    );
   578→
   579→    // Maturity gate: resolve the tier from consent + policy + flags.
   580→    const policy = await getActivePolicy();
   581→    let consentStatus: "accepted" | "rejected" | "pending" | "revoked" | null = "pending";
   582→    if (input.consentFingerprint) {
   583→      const consent = await getConsent(input.consentFingerprint);
   584→      consentStatus = (consent?.status as "accepted" | "rejected" | "pending" | "revoked" | null) ?? "pending";
   585→    }
   586→    const matureSignal = promptHasMatureSignal(input.prompt);
   587→    const tier = resolveMaturityTier({
   588→      consentStatus,
   589→      policy,
   590→      safetyFlags: safety.flags,
   591→      promptHasMatureSignal: matureSignal,
   592→    });
   593→
   594→    // Enforce min safety score as a secondary gate.
   595→    // M3: artistic-override lowers the effective min score to 30 (from 60 default)
   596→    // for artistic/editorial work. The HARD blocklist always applies — only the
   597→    // tunable min-score threshold is relaxed.
   598→    const effectiveMinScore = input.artisticOverride ? Math.min(30, policy.minSafetyScore) : policy.minSafetyScore;
   599→    const belowMinScore = safety.score < effectiveMinScore && safety.riskLevel !== "safe";
   600→
   601→    if (tier === "blocked" || belowMinScore) {
   602→      const blockReason = tier === "blocked"
   603→        ? `Maturity gate blocked this request (tier=blocked). Flags: ${safety.flags.join(",") || "none"}. Mature-enabled=${policy.matureEnabled}, consent=${consentStatus}.`
   604→        : `Safety score ${safety.score} below minimum ${effectiveMinScore}${input.artisticOverride ? " (artistic override applied)" : ""} (risk ${safety.riskLevel}).`;
   605→      await db.generation.update({
   606→        where: { id: gen.id },
   607→        data: {
   608→          status: "failed",
   609→          verdict: "rejected",
   610→          maturityTier: tier,
   611→          timings: JSON.stringify(timings),
   612→          errorMessage: blockReason,
   613→        },
   614→      });
   615→      await logEvent("policy_block", blockReason, "error", gen.id);
   616→      timings.output = elapsed(promptStart) - (timings.st3gg ?? 0);
   617→      if (timings.output < 0) timings.output = 0;
   618→      return {
   619→        id: gen.id,
   620→        status: "blocked",
   621→        imagePath: null,
   622→        safety,
   623→        judge: null,
   624→        evidence: null,
   625→        timings,
   626→        verdict: "rejected",
   627→        overallScore: null,
   628→        calibration,
   629→        loraIds: input.loraIds,
   630→        maturityTier: tier,
   631→        blockReason,
   632→        errorMessage: blockReason,
   633→        engineId: input.engineId ?? null,
   634→        backend: null,
   635→        backendMismatch: false,
   636→      };
   637→    }
   638→
   639→    // Stage: FLUX image generation (Modal GPU) with calibration
   640→    const flux = await stageFlux(input.prompt, input.style, input.aspect, input.wardrobe, gen.id, calibration, input.loraIds, input.loraWeights ?? {}, input.engineId);
   641→    timings.flux = flux.ms;
   642→    await db.generation.update({
   643→      where: { id: gen.id },
   644→      data: { imagePath: flux.imagePath, imageData: flux.imageBase64, maturityTier: tier },
   645→    });
   646→    await logEvent(
   647→      "stage_complete",
   648→      `FLUX image generated via ${flux.backend.toUpperCase()} (${flux.ms}ms) · tier=${tier}`,
   649→      "success",
   650→      gen.id
   651→    );
   652→
   653→    // Stage: Visual judge (Gemma 4 12B uncensored)
   654→    const judge = await stageJudge(flux.imagePath, input.prompt, input.style, input.wardrobe, input.brainId);
   655→    timings.judge = judge.stageMs;
   656→    await db.judgeReport.create({
   657→      data: {
   658→        generationId: gen.id,
   659→        promptAdherence: judge.promptAdherence,
   660→        visualQuality: judge.visualQuality,
   661→        aestheticScore: judge.aestheticScore,
   662→        safetyScore: judge.safetyScore,
   663→        wardrobeMatch: judge.wardrobeMatch,
   664→        overallScore: judge.overallScore,
   665→        verdict: judge.verdict,
   666→        observations: JSON.stringify(judge.observations),
   667→        strengths: JSON.stringify(judge.strengths),
   668→        weaknesses: JSON.stringify(judge.weaknesses),
   669→        stageMs: judge.stageMs,
   670→      },
   671→    });
   672→    await logEvent(
   673→      "stage_complete",
   674→      `Gemma 4 judged: ${judge.verdict} (overall ${judge.overallScore})`,
   675→      judge.verdict === "approved" ? "success" : judge.verdict === "rejected" ? "warn" : "info",
   676→      gen.id
   677→    );
   678→
   679→    // Stage: Nemotron evidence parse
   680→    const nemEngineName = flux.backendMismatch
   681→      ? "FLUX.1-schnell (Modal H100 — selected engine not deployed)"
   682→      : getEngine(input.engineId).name;
   683→    const nem = await stageNemotron(input.prompt, input.style, input.aspect, input.wardrobe, safety, judge, input.brainId, flux.backend, nemEngineName);
   684→    timings.nemotron = nem.ms;
   685→    await logEvent("stage_complete", `Nemotron structured evidence parsed (${nem.ms}ms)`, "success", gen.id);
   686→
   687→    timings.prompt = elapsed(promptStart) - (timings.flux + timings.st3gg + timings.judge + timings.nemotron);
   688→    if (timings.prompt < 0) timings.prompt = 0;
   689→    timings.output = 1;
   690→
   691→    await db.generation.update({
   692→      where: { id: gen.id },
   693→      data: {
   694→        status: "completed",
   695→        verdict: judge.verdict,
   696→        overallScore: judge.overallScore,
   697→        evidence: JSON.stringify(nem.evidence),
   698→        timings: JSON.stringify(timings),
   699→      },
   700→    });
   701→    await logEvent(
   702→      "pipeline_complete",
   703→      `Pipeline complete — verdict: ${judge.verdict}, score: ${judge.overallScore}, tier: ${tier}`,
   704→      "success",
   705→      gen.id
   706→    );
   707→
   708→    return {
   709→      id: gen.id,
   710→      status: "completed",
   711→      imagePath: `/api/image/${gen.id}`, // DB-backed URL — survives dev server restarts
   712→      safety,
   713→      judge,
   714→      evidence: nem.evidence,
   715→      timings,
   716→      verdict: judge.verdict,
   717→      overallScore: judge.overallScore,
   718→      calibration,
   719→      loraIds: input.loraIds,
   720→      maturityTier: tier,
   721→      blockReason: null,
   722→      errorMessage: null,
   723→      engineId: flux.engineId,
   724→      backend: flux.backend,
   725→      backendMismatch: flux.backendMismatch,
   726→    };
   727→  } catch (err) {
   728→    const message = err instanceof Error ? err.message : String(err);
   729→    timings.output = elapsed(promptStart);
   730→    await db.generation.update({
   731→      where: { id: gen.id },
   732→      data: {
   733→        status: "failed",
   734→        timings: JSON.stringify(timings),
   735→        errorMessage: message,
   736→      },
   737→    });
   738→    await logEvent("error", `Pipeline failed: ${truncate(message, 200)}`, "error", gen.id);
   739→    return {
   740→      id: gen.id,
   741→      status: "failed",
   742→      imagePath: null,
   743→      safety: null,
   744→      judge: null,
   745→      evidence: null,
   746→      timings,
   747→      verdict: null,
   748→      overallScore: null,
   749→      calibration,
   750→      loraIds: input.loraIds,
   751→      maturityTier: null,
   752→      blockReason: null,
   753→      errorMessage: message,
   754→      engineId: input.engineId ?? null,
   755→      backend: null,
   756→      backendMismatch: false,
   757→    };
   758→  }
   759→}
   760→
   761→export async function logEvent(
   762→  kind: string,
   763→  message: string,
   764→  severity: "info" | "warn" | "error" | "success" = "info",
   765→  generationId: string | null = null,
   766→  meta: Record<string, unknown> | null = null
   767→) {
   768→  try {
   769→    await db.auditEvent.create({
   770→      data: {
   771→        kind,
   772→        message,
   773→        severity,
   774→        generationId,
   775→        meta: meta ? JSON.stringify(meta) : null,
   776→      },
   777→    });
   778→  } catch {
   779→    // never let logging fail the pipeline
   780→  }
   781→}
   782→
   783→function truncate(s: string, n: number): string {
   784→  return s.length > n ? s.slice(0, n - 1) + "…" : s;
   785→}
